    <style>

        body {

            font-family: Arial, sans-serif;

        }

        .language-selector {

            position: absolute;

        }

            text-align: center;

        }

        select {

            padding: 5px;

            font-size: 16px;

        }

    </style>

    <script>

        const translations = {

            en: {

                welcome: "Welcome",

                description: "Select a language to get started.",

                submit: "Submit"

            },

            da: {

                welcome: "Velkommen",

                description: "Vælg et sprog for at komme i gang.",

                submit: "Indsend"

            }

        };


        function changeLanguage() {

            const language = document.getElementById("language").value;

            document.getElementById("welcome-text").innerText = translations[language].welcome;

            document.getElementById("description-text").innerText = translations[language].description;

            document.getElementById("submit-button").innerText = translations[language].submit;

        }

    </script>

</head>

<body>

    <div class="language-selector">

        <select id="language" onchange="changeLanguage()">

            <option value="en">English</option>

            <option value="da">Dansk</option>

        </select>

    </div>


    <div class="content">

        <h1 id="welcome-text">Welcome</h1>

        <p id="description-text">Select a language to get started.</p>

        <button id="submit-button">Submit</button>

    </div>
